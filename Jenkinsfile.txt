pipeline {
    agent any 
    stages {
        stage('clone repo') { 
            steps {
                sh "rm -rf My-app"
                sh "git clone https://github.com/Laxmanprashanth1/My-app.git" 
                sh "mvn clean /f My-app"
            }
        }
        stage('Test') { 
            steps {
                sh "mvn test /f My-app"
            }
        }
        stage('Deploy') { 
            steps {
                sh "mvn package /f My-app"
            }
        }
    }
}
